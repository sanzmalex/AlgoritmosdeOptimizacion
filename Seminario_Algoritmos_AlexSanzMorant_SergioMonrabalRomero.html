lista_operaciones=["+","-","*","/"]
lista_numeros=[1,2,3,4,5,6,7,8,9]

def operaciones_para(target):
  #Se crea una copia de las listas anteriores
  temp_ops = lista_operaciones[:]
  temp_nums = lista_numeros[:]

  def buscarDigito(stage, actualExpr):

    #Pequeña poda si ya es imposible llegar al valor cuando queda solo "+" y "-"
    if stage > 0:
      solo_mas_menos = True
      for op in temp_ops: #M operaciones
        if op != "+" and op != "-":#Operaciones constantes
          solo_mas_menos = False
          break
      if solo_mas_menos:
        actualVal = eval(actualExpr)
        s = 0
        for x in temp_nums:#N operaciones
          s += x
        if target < actualVal - s or target > actualVal + s:#Operaciones constantes
          return None

    #Si estamos realizando el último paso evaluamos la expresión
    if stage == 5: #Operaciones constantes
      if eval(actualExpr) == target:
        return actualExpr
      return None

    #Se aplica Backtracking para obtener los números y las operaciones
    for i in range(len(temp_nums)): #N operaciones
      actual_dig = temp_nums.pop(i)
      if stage == 0:
          ans = buscarDigito(stage + 1, str(actual_dig))
      else:
        ans = None
        for j in range(len(temp_ops)): #M operaciones
          actual_op = temp_ops.pop(j) #Operaciones constantes

          ans = buscarDigito(stage + 1, actualExpr + actual_op + str(actual_dig))

          temp_ops.insert(j, actual_op) #Operaciones constantes
          if ans is not None: #Operaciones constantes
            break

      temp_nums.insert(i, actual_dig) #Operaciones constantes
      if ans is not None: #Operaciones constantes
        return ans

    return None

  return buscarDigito(0, "")
